MOTION SIMULATOR

Welcome to the Projectile Motion Simulator!

Enter your preferred launch angle and initial velocity to observe the path of a projectile in motion. Adjust settings to explore different trajectories and see how they affect the motion. Have fun experimenting!

import math from js import document, window, console from pyodide.ffi import create_proxy from js import requestAnimationFrame, cancelAnimationFrame # Physical and display constants GRAVITY = 9.81 # Acceleration due to gravity in m/sÂ² SCALE = 10 # Pixels per meter for display scaling MARGIN = 50 # Margin in pixels for rulers and padding MAX_VELOCITY = 30 # Maximum allowed initial velocity in m/s animation_id = None # Stores the current animation frame ID start_time = None # Tracks the start time of the animation trajectory_points = [] # Stores points for drawing the trajectory path def display_error(message): """Display an error message to the user.""" error_div = document.getElementById('error-message') error_div.textContent = message def clear_error(): """Clear any displayed error messages.""" error_div = document.getElementById('error-message') error_div.textContent = "" def get_canvas(): """Get the canvas element from the DOM.""" return document.getElementById('simulation-canvas') def get_inputs(): """ Retrieve and validate user inputs for velocity and angle. Returns: tuple: (velocity, angle) if valid, (None, None) if invalid """ try: velocity_input = document.getElementById('velocity') angle_input = document.getElementById('angle') velocity = float(velocity_input.value) angle = float(angle_input.value) # Input validation if velocity < 0: raise ValueError("Velocity must be positive") if velocity > MAX_VELOCITY: # Added velocity limit check raise ValueError(f"Oops! Your initial velocity is too high. Please enter a value of {MAX_VELOCITY} m/s or lower. This limit is in place to ensure the entire projectile motion is visible within the canvas.") if angle < 0 or angle > 90: raise ValueError("Angle must be between 0 and 90 degrees") return velocity, angle except ValueError as e: display_error(str(e)) return None, None def calculate_position(t, v0, angle): """ Calculate the x and y positions of the projectile at time t. Args: t (float): Time in seconds v0 (float): Initial velocity in m/s angle (float): Launch angle in degrees Returns: tuple: (x, y) position in meters """ angle_rad = math.radians(angle) x = v0 * math.cos(angle_rad) * t y = v0 * math.sin(angle_rad) * t - 0.5 * GRAVITY * t * t return x, y def draw_rulers(ctx, canvas): """ Draw the coordinate axes and measurement rulers. Args: ctx: Canvas 2D context canvas: Canvas element """ # Draw vertical axis ctx.beginPath() ctx.moveTo(MARGIN, MARGIN) ctx.lineTo(MARGIN, canvas.height - MARGIN) ctx.stroke() # Draw horizontal axis ctx.beginPath() ctx.moveTo(MARGIN, canvas.height - MARGIN) ctx.lineTo(canvas.width - MARGIN, canvas.height - MARGIN) ctx.stroke() # Set text properties for labels ctx.font = '12px Arial' ctx.textAlign = 'right' ctx.textBaseline = 'middle' # Draw vertical ruler markings and labels height_meters = (canvas.height - 2 * MARGIN) / SCALE for i in range(0, int(height_meters) + 1, 5): y_pos = canvas.height - MARGIN - (i * SCALE) ctx.beginPath() ctx.moveTo(MARGIN - 5, y_pos) ctx.lineTo(MARGIN, y_pos) ctx.stroke() ctx.fillText(f"{i}m", MARGIN - 8, y_pos) # Draw horizontal ruler markings and labels ctx.textAlign = 'center' ctx.textBaseline = 'top' width_meters = (canvas.width - 2 * MARGIN) / SCALE for i in range(0, int(width_meters) + 1, 10): x_pos = MARGIN + (i * SCALE) ctx.beginPath() ctx.moveTo(x_pos, canvas.height - MARGIN) ctx.lineTo(x_pos, canvas.height - MARGIN + 5) ctx.stroke() ctx.fillText(f"{i}m", x_pos, canvas.height - MARGIN + 8) def draw_grid(ctx, canvas): """ Draw the background grid for better visualization. Args: ctx: Canvas 2D context canvas: Canvas element """ ctx.strokeStyle = '#eb815299' ctx.lineWidth = 0.5 # Draw vertical grid lines width_meters = (canvas.width - 2 * MARGIN) / SCALE for i in range(0, int(width_meters) + 1, 5): x_pos = MARGIN + (i * SCALE) ctx.beginPath() ctx.moveTo(x_pos, MARGIN) ctx.lineTo(x_pos, canvas.height - MARGIN) ctx.stroke() # Draw horizontal grid lines height_meters = (canvas.height - 2 * MARGIN) / SCALE for i in range(0, int(height_meters) + 1, 5): y_pos = canvas.height - MARGIN - (i * SCALE) ctx.beginPath() ctx.moveTo(MARGIN, y_pos) ctx.lineTo(canvas.width - MARGIN, y_pos) ctx.stroke() def draw_trajectory(): """Draw the path of the projectile using stored trajectory points.""" canvas = get_canvas() ctx = canvas.getContext('2d') ctx.beginPath() ctx.strokeStyle = '#6f598266' # Semi-transparent purple ctx.lineWidth = 2 # Connect all trajectory points with a line for i, point in enumerate(trajectory_points): if i == 0: ctx.moveTo(point[0] * SCALE + MARGIN, canvas.height - (point[1] * SCALE + MARGIN)) else: ctx.lineTo(point[0] * SCALE + MARGIN, canvas.height - (point[1] * SCALE + MARGIN)) ctx.stroke() def draw_scene(x, y): """ Draw the complete scene including grid, rulers, trajectory, and projectile. Args: x (float): Current x position of projectile in meters y (float): Current y position of projectile in meters """ try: canvas = get_canvas() ctx = canvas.getContext('2d') # Clear previous frame ctx.clearRect(0, 0, canvas.width, canvas.height) # Set line properties ctx.lineWidth = 1 ctx.strokeStyle = '#6f5982' # Draw scene elements draw_grid(ctx, canvas) draw_rulers(ctx, canvas) draw_trajectory() # Draw the projectile as a purple circle ctx.beginPath() ctx.arc(x * SCALE + MARGIN, canvas.height - (y * SCALE + MARGIN), 5, 0, 2 * math.pi) ctx.fillStyle = '#6f5982' ctx.fill() except Exception as e: console.log(f"Error in draw_scene: {str(e)}") display_error(f"Error drawing scene: {str(e)}") def animation_frame(timestamp): """ Animation loop callback function. Args: timestamp: Current time provided by requestAnimationFrame """ global start_time, animation_id, trajectory_points try: # Initialize start time on first frame if start_time is None: start_time = timestamp elapsed_time = (timestamp - start_time) / 1000 # Convert to seconds v0, angle = get_inputs() if v0 is None or angle is None: return # Calculate current position x, y = calculate_position(elapsed_time, v0, angle) canvas = get_canvas() # Record trajectory point trajectory_points.append((x, y)) # Check if simulation should end if y < 0 or x * SCALE > (canvas.width - 2 * MARGIN): cancelAnimationFrame(animation_id) start_time = None draw_scene(x, max(y, 0)) return # Continue animation draw_scene(x, y) animation_id = requestAnimationFrame(create_proxy(animation_frame)) except Exception as e: console.log(f"Error in animation_frame: {str(e)}") display_error(f"Animation error: {str(e)}") def start_simulation(*args): """Start or restart the simulation.""" global start_time, animation_id, trajectory_points try: clear_error() v0, angle = get_inputs() if v0 is None or angle is None: return # Reset simulation state trajectory_points = [] # Cancel any existing animation if animation_id is not None: cancelAnimationFrame(animation_id) # Start new animation start_time = None animation_id = requestAnimationFrame(create_proxy(animation_frame)) except Exception as e: console.log(f"Error in start_simulation: {str(e)}") display_error(f"Error starting simulation: {str(e)}") def init(): """Initialize the canvas with grid and rulers.""" try: canvas = get_canvas() ctx = canvas.getContext('2d') # Draw initial grid and rulers draw_grid(ctx, canvas) draw_rulers(ctx, canvas) except Exception as e: console.log(f"Error in init: {str(e)}") display_error(f"Error initializing canvas: {str(e)}") # Initialize the simulation when the page loads init()
